def add_student():
    try:
        roll_no = int(input("Enter Roll No: "))
        name = input("Enter Name: ").strip()
        branch = input("Enter Branch: ").strip()
        year = input("Enter Year: ").strip()
        gender = input("Enter Gender (M/F): ").strip().upper()
        age = int(input("Enter Age: "))
        attendance = float(input("Enter Attendance % (0-100): "))
        mid1 = float(input("Enter Mid1 Marks: "))
        mid2 = float(input("Enter Mid2 Marks: "))
        quiz = float(input("Enter Quiz Marks: "))
        final = float(input("Enter Final Marks: "))

        if not (0 <= attendance <= 100):
            print(" Attendance must be between 0 and 100")
            return

        with open(CSV_FILE, "r") as file:
            reader = csv.DictReader(file)
            for row in reader:
                if int(row["Roll_No"]) == roll_no:
                    print("Student with this Roll No already exists.")
                    return

        with open(CSV_FILE, "a", newline="") as file:
            writer = csv.writer(file)
            writer.writerow([roll_no, name, branch, year, gender, age,
                             attendance, mid1, mid2, quiz, final])
        print("Student added successfully!")

    except ValueError:
        print("Please enter valid data (numbers where required).")


def search_student():
    choice = input("Search by (1) Roll_No or (2) Name: ").strip()

    found = False
    with open(CSV_FILE, "r") as file:
        reader = csv.DictReader(file)
        if choice == "1":
            try:
                roll_no = int(input("Enter Roll_No: "))
                for row in reader:
                    if int(row["Roll_No"]) == roll_no:
                        print("Student Found:")
                        print(row)
                        found = True
                        break
            except ValueError:
                print("Roll_No must be a number.")
                return

        elif choice == "2":
            name_part = input("Enter Name (partial allowed): ").strip().lower()
            for row in reader:
                if name_part in row["Name"].lower():
                    if not found:
                        print("Matches Found:")
                    print(row)
                    found = True
        else:
            print("Invalid option. Choose 1 or 2.")
            return

    if not found:
        print("No matching record found.")

def delete_student():
    try:
        roll_no = int(input("Enter Roll_No of student to delete: "))
        students = []

        with open(CSV_FILE, "r") as file:
            reader = csv.DictReader(file)
            for row in reader:
                students.append(row)

        found = False
        for row in students:
            if int(row["Roll_No"]) == roll_no:
                found = True
                print("Student found:")
                print(row)

                confirm = input("Are you sure you want to delete this student? (Y/N): ").strip().upper()
                if confirm == "Y":
                    students.remove(row)

                    # Save deleted record into students_deleted.csv
                    deleted_file = os.path.join(os.path.dirname(CSV_FILE), "students_deleted.csv")
                    file_exists = os.path.exists(deleted_file)
                    with open(deleted_file, "a", newline="") as df:
                        writer = csv.DictWriter(df, fieldnames=row.keys())
                        if not file_exists:
                            writer.writeheader()
                        writer.writerow(row)

                    # Rewrite updated students.csv
                    with open(CSV_FILE, "w", newline="") as file:
                        fieldnames = ["Roll_No", "Name", "Branch", "Year", "Gender", "Age",
                                      "Attendance_%", "Mid1_Marks", "Mid2_Marks", "Quiz_Marks", "Final_Marks"]
                        writer = csv.DictWriter(file, fieldnames=fieldnames)
                        writer.writeheader()
                        writer.writerows(students)

                    print("Student deleted successfully (moved to students_deleted.csv).")
                else:
                    print("Deletion cancelled.")
                break

        if not found:
            print("Student with this Roll_No not found.")

    except ValueError:
        print("Invalid input. Roll_No must be a number.")
def generate_report():
    branch = input("Enter Branch: ").strip()
    year = input("Enter Year: ").strip()

    students = []
    with open(CSV_FILE, "r") as file:
        reader = csv.DictReader(file)
        for row in reader:
            if row["Branch"].lower() == branch.lower() and row["Year"].lower() == year.lower():
                students.append(row)

    if not students:
        print("No students found for this Branch and Year.")
        return
    for row in students:
        try:
            row["Final_Marks"] = float(row["Final_Marks"])
        except ValueError:
            row["Final_Marks"] = 0.0

    total_students = len(students)
    avg_marks = sum(s["Final_Marks"] for s in students) / total_students
    top_student = max(students, key=lambda x: x["Final_Marks"])
    low_student = min(students, key=lambda x: x["Final_Marks"])

    grades = {"A": 0, "B": 0, "C": 0, "D": 0}
    for s in students:
        marks = s["Final_Marks"]
        if marks >= 90:
            grades["A"] += 1
        elif marks >= 75:
            grades["B"] += 1
        elif marks >= 50:
            grades["C"] += 1
        else:
            grades["D"] += 1

    print("\n--- Report ---")
    print(f"Branch: {branch}, Year: {year}")
    print(f"Total Students: {total_students}")
    print(f"Class Average: {avg_marks:.2f}")
    print(f"Topper: {top_student['Name']} ({top_student['Final_Marks']})")
    print(f"Lowest: {low_student['Name']} ({low_student['Final_Marks']})")
    print("Grade Distribution:", grades)

    report_file = os.path.join(os.path.dirname(CSV_FILE), f"report_{branch}_{year}.csv")
    with open(report_file, "w", newline="") as file:
        writer = csv.writer(file)
        writer.writerow(["Branch", "Year", "Total Students", "Average Marks", "Topper", "Topper Marks",
                         "Lowest", "Lowest Marks", "Grade A", "Grade B", "Grade C", "Grade D"])
        writer.writerow([branch, year, total_students, round(avg_marks, 2),
                         top_student["Name"], top_student["Final_Marks"],
                         low_student["Name"], low_student["Final_Marks"],
                         grades["A"], grades["B"], grades["C"], grades["D"]])

    print(f"Report saved as {report_file}")

def bulk_import():
    file_path = input("Enter the path of the CSV file to import: ").strip()

    if not os.path.exists(file_path):
        print("File not found. Please check the path.")
        return

    imported_count = 0
    skipped_count = 0
    errors = []

    with open(file_path, "r") as infile, open(CSV_FILE, "a", newline="") as outfile:
        reader = csv.DictReader(infile)
        writer = csv.writer(outfile)

        existing_rolls = set()
        with open(CSV_FILE, "r") as f:
            existing_reader = csv.DictReader(f)
            for row in existing_reader:
                existing_rolls.add(int(row["Roll_No"]))

        for i, row in enumerate(reader, start=2):  # start=2 because line 1 is header
            try:
                roll_no = int(row["Roll_No"])
                if roll_no in existing_rolls:
                    errors.append((i, "Duplicate Roll_No"))
                    skipped_count += 1
                    continue

                name = row["Name"].strip()
                branch = row["Branch"].strip()
                year = row["Year"].strip()
                gender = row["Gender"].strip().upper()
                age = int(row["Age"])
                attendance = float(row["Attendance_%"])
                mid1 = float(row["Mid1_Marks"])
                mid2 = float(row["Mid2_Marks"])
                quiz = float(row["Quiz_Marks"])
                final = float(row["Final_Marks"])

                if not (0 <= attendance <= 100):
                    errors.append((i, "Invalid Attendance"))
                    skipped_count += 1
                    continue

                writer.writerow([roll_no, name, branch, year, gender, age,
                                 attendance, mid1, mid2, quiz, final])
                existing_rolls.add(roll_no)
                imported_count += 1

            except Exception as e:
                errors.append((i, str(e)))
                skipped_count += 1

    if errors:
        error_file = os.path.join(os.path.dirname(CSV_FILE), "import_errors.csv")
        with open(error_file, "w", newline="") as ef:
            ew = csv.writer(ef)
            ew.writerow(["Line", "Error"])
            ew.writerows(errors)
        print(f"Errors logged to {error_file}")

    print(f"Bulk Import Completed: {imported_count} records imported, {skipped_count} skipped.")
def sort_filter_students():
    print("\n--- Sort / Filter ---")
    print("1. Sort by Final Marks")
    print("2. Filter by Attendance < threshold")
    choice = input("Enter choice: ").strip()

    students = []
    with open(CSV_FILE, "r") as file:
        reader = csv.DictReader(file)
        for row in reader:
            try:
                row["Final_Marks"] = float(row["Final_Marks"])
            except ValueError:
                row["Final_Marks"] = 0.0
            try:
                row["Attendance_%"] = float(row["Attendance_%"])
            except ValueError:
                row["Attendance_%"] = 0.0
            students.append(row)

    if not students:
        print("No student records found.")
        return

    result = []

    if choice == "1":
        order = input("Sort order (A=Ascending, D=Descending): ").strip().upper()
        reverse = True if order == "D" else False
        result = sorted(students, key=lambda x: x["Final_Marks"], reverse=reverse)
        print("\nSorted Students (by Final Marks):")
        for s in result:
            print(f"{s['Roll_No']} - {s['Name']} - {s['Final_Marks']}")

    elif choice == "2":
        try:
            threshold = float(input("Enter Attendance threshold (0-100): "))
        except ValueError:
            print("Invalid threshold.")
            return
        result = [s for s in students if s["Attendance_%"] < threshold]
        if not result:
            print(f"No students found with attendance below {threshold}.")
            return
        print(f"\nStudents with Attendance < {threshold}:")
        for s in result:
            print(f"{s['Roll_No']} - {s['Name']} - {s['Attendance_%']}%")

    else:
        print("Invalid choice.")
        return

    export = input("Do you want to export this result to CSV? (Y/N): ").strip().upper()
    if export == "Y" and result:
        export_file = os.path.join(os.path.dirname(CSV_FILE), "sort_filter_result.csv")
        with open(export_file, "w", newline="") as f:
            writer = csv.DictWriter(f, fieldnames=result[0].keys())
            writer.writeheader()
            writer.writerows(result)
        print(f"Results exported to {export_file}")
